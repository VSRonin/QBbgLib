149  const T* const getResult(quint32 group, qint64 id) const
150  {
151  static_assert(std::is_base_of<QBbgAbstractResponse, T>::value, "getResult Template type must be derived from QBbgAbstractResponse");
152  const QBbgAbstractResponse* const genericRes = getResultGeneric(group, id);
153  if (!genericRes)
154  return nullptr;
155  if (!std::is_same<QBbgAbstractResponse, T>::value) {
156  switch (genericRes->responseType()) {
157  case QBbgAbstractResponse::ResponseType::ReferenceDataResponse:
158  if (!std::is_same<QBbgReferenceDataResponse, T>::value)
159  return nullptr;
160  break;
161  case QBbgAbstractResponse::ResponseType::HistoricalDataResponse:
162  if (!std::is_same<QBbgHistoricalDataResponse, T>::value)
163  return nullptr;
164  break;
165  case QBbgAbstractResponse::ResponseType::PortfolioDataResponse:
166  if (!std::is_same<QBbgPortfolioDataResponse, T>::value)
167  return nullptr;
168  break;
169  case QBbgAbstractResponse::ResponseType::IntraDayTickResponse:
170  if (!std::is_same<QBbgIntradayTickResponse, T>::value)
171  return nullptr;
172  break;
173  default:
174  Q_UNREACHABLE(); //Unhandled response type
175  }
176  }
177  Q_ASSERT(dynamic_cast<const T* const>(genericRes));
178  return static_cast<const T* const>(genericRes);
179  }

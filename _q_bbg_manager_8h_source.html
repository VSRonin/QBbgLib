125  const QBbgAbstractResponse* const genericRes = getResultGeneric(group, id);
126  if (!genericRes)
127  return nullptr;
128  if (!std::is_same<QBbgAbstractResponse, T>::value) {
129  switch (genericRes->responseType()) {
130  case QBbgAbstractResponse::ResponseType::ReferenceDataResponse:
131  if (!std::is_same<QBbgReferenceDataResponse, T>::value)
132  return nullptr;
133  break;

137  break;

140  return nullptr;
141  break;
142  default:
143  Q_UNREACHABLE(); //Unhandled response type
144  }
145  }
146  return static_cast<const T* const>(genericRes);
